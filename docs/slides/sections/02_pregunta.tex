% Pregunta de Investigación
\section{Pregunta de Investigación}

\begin{frame}
\frametitle{Pregunta Central}
\begin{center}
\Large\textbf{¿Cómo podría estructurarse el recaudo tributario del Estado colombiano bajo distintos escenarios de legalización de la distribución de marihuana?}
\end{center}
\vspace{1cm}
\normalsize
Subpreguntas específicas:
\begin{enumerate}
    \item ¿Cuál sería la base potencial de contribuyentes por perfil sociodemográfico?
    \item ¿Qué estructura tributaria maximiza ingresos fiscales?
    \item ¿Cómo varían los ingresos bajo diferentes marcos regulatorios?
\end{enumerate}
\end{frame}

\begin{frame}
\frametitle{Objetivos}
\textbf{Objetivo General:}
\begin{itemize}
    \item Construir un modelo predictivo de recaudos tributarios bajo distintos escenarios de legalización
\end{itemize}

\vspace{0.5cm}
\textbf{Objetivos Específicos:}
\begin{enumerate}
    \item Procesar y analizar datos sociodemográficos de encuestas nacionales
    \item Identificar determinantes de demanda y disposición a pagar
    \item Simular escenarios fiscales con diferentes tasas y tipos impositivos
    \item Proyectar ingresos tributarios potenciales
\end{enumerate}
\end{frame}
